a)
def factorial_loop(n):

	sum = 1
	if n == 0:
		return sum
	else:
		for i in range(1, n+1):
			sum *= i
		return sum
b)	
def nchoosek(n, k):
	sum = factorial_loop(n)/((factorial_loop(k)) * factorial_loop(n - k))
	return sum

def Pascal(n):
	for i in range(0, n):
		for j in range(0, i+1):
			print(int(nchoosek(i, j)), end=' ')
		print("\n")
 		
	
n = int(input("Skriv inn n"))
